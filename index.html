# Define a function to compute the expected profit analytically
compute_expected_profit <- function(n, p = 0.98, seats = 100, r, c) {
  # Create a vector of possible arrivals from 0 to n
  x_vals <- 0:n
  
  # Compute the probability for each number of arrivals (Binomial PMF)
  probs <- dbinom(x_vals, size = n, prob = p)
  
  # Compute profit based on whether x <= seats or x > seats
  profit <- ifelse(x_vals <= seats, 
                   r * x_vals,                # When x <= seats: Profit = r * x
                   seats * r - c * (x_vals - seats))  # When x > seats: Profit = 100r - c*(x-100)
  
  # Expected profit is the weighted sum of profit over the binomial probabilities
  expected_profit <- sum(profit * probs)
  sd_value <- sqrt(sum((profit - expected_profit)^2 * probs))  # Calculate standard deviation
  
  return(c(expected_profit, sd_value))
}

# Example parameter values
r <- 200   # revenue per passenger (ticket price)
c <- 500   # compensation cost per bumped passenger

# Initialize vectors to store results
n_values <- 90:110
expected_profits <- numeric(length(n_values))
sd_values <- numeric(length(n_values))

# Create a plot with error bars
plot(n_values, expected_profits, type = "b", col = "blue", pch = 19,
     xlab = "Number of Reservations (n)", ylab = "Expected Profit",
     main = "Expected Profit vs. Number of Reservations",
     ylim = range(expected_profits - sd_values, expected_profits + sd_values) * c(0.95, 1.05))  # Adjust y-limits for better visibility

# Add error bars to the plot
arrows(n_values, expected_profits - sd_values, n_values, expected_profits + sd_values,
       angle = 90, code = 3, length = 0.1, col = "red")
